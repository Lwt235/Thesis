    \For{$\a \in \omega$}{
      $Eb^i_{j,a} = e^{\lambda_i*R^i_{j,a}}+e^{\mu_i*len^i_{j,a}}-2$\\

\section{剩余工人分配算法}
\renewcommand{\algorithmcfname}{算法}
\renewcommand{\thealgocf}{4-4}
\begin{algorithm}[H]

  \SetAlgoNoLine %去掉横线
  \SetKwInput{KwData}{辅助数组}
  \SetKwInOut{Input}{输入}
  \SetKwInOut{Output}{输出}
  \SetKwData{u}{$\tau^i_u$}
  \SetKwData{v}{$\omega^i_v$}
  \SetKwData{W}{$W^i$}

  \Input{ 初始任务图$G^i$,信任工人分配图$G^i_t$,第i次任务$T^i$,成本因子$Cf_i$,分辨强度$Re_i$,竞争权重集合$W^i$ }
  \Output{ 工人任务分配图$G^i_f$ }
  \KwData{ 匹配数组$match$ }

  $G^i_f \leftarrow G^i_t$\\
  \For{$\u \in T^i$}{
    $initial$ $\omega^i_u$ $as$ $an$ $empty$ $set$\\
    \For{$\v \in G^i[\u]$ $is$ $not$ $matched$}{
      $add$ \v $to$ $\omega^i_u$\\
    }
    \uIf{$G^i_t[\u]$ $is$ $empty$}{
      \For{ $\v \in \omega^i_u$ }{
        $G^i_f[\v] \leftarrow \u$\\
      }
    }
    \Else{
      $sort$ $\omega^i_u$ $by$ \W $from$ $largest$ $to$ $smallest$\\
      $let$ $L$ $equals$ $to$ $the$ $length$ $of$ $T_u$\\
      \For{ $\v \in \omega^i_u$ }{
        \If{ $cnt > \frac{Re_i}{Re_i+Cf_i} * L$ }{ $break$ }
        $G^i_f[\v] \leftarrow \u$\\
        $cnt = cnt + 1$\\
      }
    }
  }
  \caption{剩余工人分配算法}
\end{algorithm}


\section{RDD算法}
\renewcommand{\algorithmcfname}{算法}
\renewcommand{\thealgocf}{4-5}
\begin{algorithm}[H]

  \SetAlgoNoLine %去掉横线
  \SetKwInput{KwData}{辅助数组}
  \SetKwInOut{Input}{输入}
  \SetKwInOut{Output}{输出}
  \SetKwData{u}{$\omega^t_u$}
  \SetKwData{v}{$\tau_v$}
  \SetKwData{W}{$W^i_j$}

  \Input{ 数据集合$Da^i_j$,迭代阈值$e$ }
  \Output{ 标准数据$Ds^i$ }
  \KwData{ 权重数组$w^i$ }

  $Ds^i = Da^i_1$\\
  $Ds^i_f = Ds^i + e$\\
  \While{$\lvert Ds^i-Ds^i_f \rvert \geq e $}{
    $ S^i \leftarrow \sum d(Da^i_{j,a},Ds^i) $\\
    \For{$Da^i_{j,a} \in Da^i_j$}{
      $ w^i_a = \log{S^i} - \log{d(Da^i_{j,a},Ds^i)} $\\
    }
    $ Ds^i_f \leftarrow Ds^i $\\
    $ Ds^i = \frac{\sum w^i_a*Da^i_{j,a}}{\sum w^i_a} $\\
  }

  \caption{RDD算法(Real Data Discovery)}
\end{algorithm}


\section{信任度更新算法}
\renewcommand{\algorithmcfname}{算法}
\renewcommand{\thealgocf}{4-6}
\begin{algorithm}[H]

  \SetAlgoNoLine %去掉横线
  \SetKwInput{KwData}{辅助数组}
  \SetKwInOut{Input}{输入}
  \SetKwInOut{Output}{输出}

  \Input{ 个人提交数据$Da^i_{j,a}$,偏差阈值$D^i_m$,低强度因子$f_l$,高强度因子$f_h$(,标准数据$Ds^i_j$) }
  \Output{ 信任度更新$tr_a$ }

  \uIf{$Ds^i_j$ $doesn't$ $exist$}{
    $Ds^i_j \leftarrow RDD(Da^i_j)$\\
    $D^i_a = \lvert \frac{Da^i_{j,a} - Ds^i_j}{Ds^i_j}  \rvert$\\
    $tr_a = f_l * e^{D^i_m-D^i_a}$
  }
  \Else{
    $D^i_a = \lvert \frac{Da^i_{j,a} - Ds^i_j}{Ds^i_j}  \rvert$\\
    $tr_a = f_h * e^{D^i_m-D^i_a}$
  }

  \caption{信任度更新算法}
\end{algorithm}
